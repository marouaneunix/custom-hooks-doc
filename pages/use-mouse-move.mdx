## Example

```tsx filename="demo.tsx" copy
const Target = () => {
  const {position, ref} = useMouseMove();
  
  return (
    <div data-testid="target" ref={ref}>
      {position && `${position.clientX}${position.clientY}`}
    </div>
  )
};
```
